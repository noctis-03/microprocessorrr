// ══════════════════════════════════════
// ESP‑NOW 콜백
// ══════════════════════════════════════
void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int data_len) {
  if (data_len == sizeof(Message)) {
    memcpy((void*)&rxMsg, data, sizeof(Message));
    if (rxMsg.msgType == MSG_SENSOR_DATA) {
      lastTemp = rxMsg.temperature;
      lastHumidity = rxMsg.humidity;
      if (rxMsg.year >= 2025) {
        year = rxMsg.year; month = rxMsg.month; day = rxMsg.day;
        weekday = rxMsg.weekday; hour = rxMsg.hour; minute = rxMsg.minute;
        timeEverSynced = true;
        // 마지막 동기화 시각 기록
        lastSyncYear = year; lastSyncMonth = month; lastSyncDay = day;
        lastSyncHour = hour; lastSyncMin = minute;
      }
      if (rxMsg.fcHour[0] >= 0) {
        memcpy(fcHour,  rxMsg.fcHour,  sizeof(fcHour));
        memcpy(fcTemp,  rxMsg.fcTemp,  sizeof(fcTemp));
        memcpy(fcHumid, rxMsg.fcHumid, sizeof(fcHumid));
        memcpy(fcCode,  rxMsg.fcCode,  sizeof(fcCode));
        fcValid = true;
      }
      todayMax = rxMsg.todayMax;
      todayMin = rxMsg.todayMin;
      todayPop = rxMsg.todayPop;
      pm25     = rxMsg.pm25;
      usAqi    = rxMsg.usAqi;
      if (usAqi > 0 || pm25 > 0) envValid = true;

      // ★ 설정값 수신: 버전이 더 높을 때만 갱신
      if (rxMsg.config.configVersion > lastCfgVersion) {
        appliedCfg = rxMsg.config;
        lastCfgVersion = rxMsg.config.configVersion;
        Serial.printf("[CFG] updated from B (ver %d)\n", lastCfgVersion);
      }

      dataReceived = true;
      everGotData = true;
    }
    return;
  }

  if (data_len == sizeof(AIChunk)) {
    AIChunk ck;
    memcpy(&ck, data, sizeof(AIChunk));
    if (ck.msgType != MSG_AI_CHUNK) return;
    if (ck.totalChunks == 0 || ck.totalChunks > AI_MAX_CHUNKS) return;
    if (ck.chunkIndex >= ck.totalChunks) return;

    aiTotalChunks = ck.totalChunks;
    aiRecvDay = ck.commentDay;

    uint8_t len = ck.dataLen;
    if (len > AI_CHUNK_SIZE) len = AI_CHUNK_SIZE;
    int off = ck.chunkIndex * AI_CHUNK_SIZE;
    if (off + len <= AI_BUF_SIZE) {
      memcpy(aiRecvBuf + off, ck.data, len);
      aiChunkGot[ck.chunkIndex] = true;
    }

    bool all = true;
    for (int i = 0; i < aiTotalChunks; i++) if (!aiChunkGot[i]) { all = false; break; }
    if (all) {
      int total = (aiTotalChunks - 1) * AI_CHUNK_SIZE + len;
      if (total >= AI_BUF_SIZE) total = AI_BUF_SIZE - 1;
      aiRecvBuf[total] = '\0';
      aiAllReceived = true;
    }
    aiChunkReceived = true;
    return;
  }
}

// ── 24x24 1bpp 비트맵을 scale배 확대해서 그리기 ──
void drawBitmap2x(int x, int y, const unsigned char* bmp, int scale) {
  for (int row = 0; row < ICON_H; row++) {
    for (int col = 0; col < ICON_W; col++) {
      int byteIndex = row * ((ICON_W + 7) / 8) + (col / 8);
      uint8_t b = pgm_read_byte(&bmp[byteIndex]);
      bool on = b & (0x80 >> (col % 8));
      if (on) {
        display.fillRect(x + col * scale, y + row * scale, scale, scale, GxEPD_BLACK);
      }
    }
  }
}

int daysInMonth(int y, int m) {
  int days[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
  if (m == 2 && ((y%4==0 && y%100!=0) || y%400==0)) return 29;

  int diff = day - 1;

}

// ══════════════════════════════════════
// ESP‑NOW 초기화 (특정 채널로)
// ══════════════════════════════════════
void setRadioChannel(int ch) {
  esp_wifi_set_promiscuous(true);
  esp_wifi_set_channel(ch, WIFI_SECOND_CHAN_NONE);
  esp_wifi_set_promiscuous(false);
}

void initESPNow() {
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(100);
  setRadioChannel(savedChannel);
  if (esp_now_init() != ESP_OK) { Serial.println("ESP-NOW FAIL"); return; }
  esp_now_register_recv_cb(onDataRecv);
  esp_now_register_send_cb(esp_now_send_cb_t(onDataSent));
  esp_now_peer_info_t peerInfo = {};
  memcpy(peerInfo.peer_addr, peerMAC_B, 6);
  peerInfo.channel = 0;       // 0 = 현재 라디오 채널 사용
  peerInfo.encrypt = false;
  esp_now_add_peer(&peerInfo);
  Serial.printf("ESP-NOW ready CH:%d\n", savedChannel);
}

// ── 특정 채널에서 한 번 요청 후 응답 대기 ──
bool tryRequestOnChannel(int ch) {
  setRadioChannel(ch);
  Message req = {};
  req.msgType = MSG_REQUEST;
  dataReceived = false;
  esp_now_send(peerMAC_B, (uint8_t *)&req, sizeof(Message));
  unsigned long start = millis();
  while (!dataReceived && (millis() - start < DATA_WAIT_TIMEOUT_MS)) delay(10);
  return dataReceived;
}

// ══════════════════════════════════════
// 센서 데이터 요청
//   1) 마지막 성공 채널(savedChannel)에서 SCAN_ATTEMPTS회 시도
//   2) 실패하면 1~13 전체 채널 스캔
//   3) 성공한 채널을 savedChannel 에 저장
//   gotFreshData: 이번에 새 데이터를 받았는지 표시
// ══════════════════════════════════════
void requestSensorData() {
  gotFreshData = false;

  // 1) 우선 마지막 성공 채널에서 시도
  for (int attempt = 0; attempt < SCAN_ATTEMPTS; attempt++) {
    if (tryRequestOnChannel(savedChannel)) {
      gotFreshData = true;
      Serial.printf("OK on saved CH:%d (try %d) T:%.1f H:%.1f%%\n",
                    savedChannel, attempt + 1, lastTemp, lastHumidity);
      return;
    }
  }

  // 2) 전체 채널 스캔
  Serial.println("Saved channel failed -> scanning 1..13");
  for (int ch = CH_MIN; ch <= CH_MAX; ch++) {
    if (ch == savedChannel) continue;   // 이미 위에서 시도함
    if (tryRequestOnChannel(ch)) {
      savedChannel = ch;                // ★ 성공한 채널 기억
      gotFreshData = true;
      Serial.printf("Found B on CH:%d  T:%.1f H:%.1f%%\n",
                    ch, lastTemp, lastHumidity);
      return;
    }
  }

  // 3) 실패: 라디오를 savedChannel 로 복구하고 캐시 데이터 사용
  setRadioChannel(savedChannel);
  Serial.println("No response on any channel -> use cached data");
}

void requestAIComment() {
  gotFreshAI = false;
  if (aiCommentValid && aiCommentDay == day) return;   // 오늘 멘트 이미 보유
  AIChunk req = {};
  req.msgType = MSG_AI_REQUEST;
  setRadioChannel(savedChannel);
  for (int attempt = 0; attempt < SCAN_ATTEMPTS; attempt++) {
    for (int i = 0; i < AI_MAX_CHUNKS; i++) aiChunkGot[i] = false;
    aiAllReceived = false; aiChunkReceived = false; aiTotalChunks = 0;
    esp_now_send(peerMAC_B, (uint8_t *)&req, sizeof(AIChunk));
    unsigned long start = millis();
    while (!aiAllReceived && (millis() - start < AI_WAIT_TIMEOUT_MS)) delay(10);
    if (aiAllReceived) {
      strncpy(aiComment, aiRecvBuf, AI_BUF_SIZE - 1);
      aiComment[AI_BUF_SIZE - 1] = '\0';
      aiCommentDay = aiRecvDay;
      aiCommentValid = true;
      gotFreshAI = true;
      Serial.printf("[AI] comment received (day %d): %s\n", aiRecvDay, aiComment);
      return;
    }
  }
  Serial.println("[AI] no comment yet");
}

const unsigned char* weatherIcon(uint8_t code) {
  if (code == 0) return icon_sun;
  if (code <= 3) return icon_cloud;
  if (code <= 48) return icon_fog;
  if (code <= 67) return icon_rain;
  if (code <= 77) return icon_snow;
  if (code <= 82) return icon_rain;
  return icon_storm;
}

const char* aqiGrade(uint8_t aqi) {
  if (aqi == 0)   return "N/A";
  if (aqi <= 50)  return "Good";
  if (aqi <= 100) return "Moderate";
  if (aqi <= 150) return "Sensitive";
  if (aqi <= 200) return "Unhealthy";
  if (aqi <= 300) return "Very Bad";
  return "Hazardous";
}

// ── 현재 시각에 가장 가까운 예보 슬롯 인덱스 ──
int pickSlotByNow() {
  if (!fcValid) return 3;
  int best = 0, bestDiff = 99;
  for (int i = 0; i < WEATHER_SLOTS; i++) {
    if (fcHour[i] < 0) continue;
    int diff = abs((int)fcHour[i] - hour);
    if (diff < bestDiff) { bestDiff = diff; best = i; }
  }
  return best;
}

// ── 워드랩 (영어) ──
void drawWrappedText(const GFXfont* font, const char* text,
                     int x, int y, int maxWidth, int lineH, int maxLines) {
  display.setFont(font);
  display.setTextColor(GxEPD_BLACK);
  char line[128] = "";
  char word[64];
  int wi = 0, lineCount = 0;
  const char* p = text;

  auto flushWord = [&](bool endOfText) {
    word[wi] = '\0';
    if (wi == 0 && !endOfText) return;
    char trial[160];
    if (line[0] == '\0') snprintf(trial, sizeof(trial), "%s", word);
    else                 snprintf(trial, sizeof(trial), "%s %s", line, word);
    int16_t bx, by; uint16_t bw, bh;
    display.getTextBounds(trial, 0, 0, &bx, &by, &bw, &bh);
    if ((int)bw > maxWidth && line[0] != '\0') {
      display.setCursor(x, y + lineCount * lineH);
      display.print(line);
      lineCount++;
      strncpy(line, word, sizeof(line) - 1); line[sizeof(line)-1] = '\0';
    } else {
      strncpy(line, trial, sizeof(line) - 1); line[sizeof(line)-1] = '\0';
    }
    wi = 0;
  };

  while (*p && lineCount < maxLines) {
    if (*p == ' ' || *p == '\n') {
      flushWord(false);
      if (*p == '\n') {
        display.setCursor(x, y + lineCount * lineH);
        display.print(line); lineCount++; line[0] = '\0';
      }
    } else {
      if (wi < (int)sizeof(word) - 1) word[wi++] = *p;
    }
    p++;
  }
  if (lineCount < maxLines) {
    flushWord(true);
    if (line[0] != '\0') {
      display.setCursor(x, y + lineCount * lineH);
      display.print(line);
    }
  }
}

// fetch=true: B와 통신 후 갱신 / fetch=false: 통신 없이 현재 데이터로 화면만 그림
void refreshDisplay(bool fetch = true) {
  if (fetch) {
    requestSensorData();
    if (currentMode == MODE_AIRENV && appliedCfg.aiEnabled) requestAIComment();
  }

  epdPowerOn();
  display.init(115200, true, 50, false);
  display.setRotation(0);
  display.setFullWindow();

  bool doFullRefresh = (partialRefreshCount >= appliedCfg.fullRefreshEvery);

  switch (currentMode) {
    case MODE_CALENDAR: if (calendarView == 0) drawDayView(); else drawMonthView(); break;
    case MODE_SENSOR:   drawWeatherView(); break;
    case MODE_AIRENV:   drawAirEnvView(); break;
    case MODE_INFO:     drawInfoView(); break;
  }

  if (doFullRefresh) { display.display(false); partialRefreshCount = 0; }
  else { display.display(true); partialRefreshCount++; }

  display.hibernate();
  epdPowerOff();
}

void enterDeepSleep() {
  esp_now_deinit();
  WiFi.mode(WIFI_OFF);
  if (currentMode != MODE_CALENDAR) {
    currentMode = MODE_CALENDAR;
    refreshDisplay(false);   // ★ 통신 없이 캘린더 화면만 다시 그림
  }
  showSleepIndicator();

  uint64_t sleepTime;
  if (timeEverSynced) {
    int wakeHour = appliedCfg.wakeHour;
    int sec;
    if (hour < wakeHour) sec = ((wakeHour - hour) * 3600) - (minute * 60);
    else                 sec = ((24 - hour + wakeHour) * 3600) - (minute * 60);
    if (sec <= 0) sec = 3600;
    sleepTime = (uint64_t)sec * 1000000ULL;
  } else {
    sleepTime = REFRESH_INTERVAL_SEC * 1000000ULL;
  }

  esp_sleep_enable_timer_wakeup(sleepTime);
  esp_sleep_enable_ext0_wakeup(GPIO_NUM_2, LOW);
  uint64_t btnMask = (1ULL << BTN_EXIT) | (1ULL << BTN_UP) |
                     (1ULL << BTN_DOWN) | (1ULL << BTN_OK);
  esp_sleep_enable_ext1_wakeup(btnMask, ESP_EXT1_WAKEUP_ANY_LOW);

  Serial.println("Deep Sleep...");
  Serial.flush();
  esp_deep_sleep_start();
}

void infoLoop() {
  resetActivityTimer();
  while (!isTimeoutReached()) {
    if (isButtonPressed(BTN_EXIT)) {
      waitButtonRelease(BTN_EXIT);
      currentMode = MODE_CALENDAR; exitReason = EXIT_MODE_CHANGE; return;
    }
    if (isButtonPressed(BTN_UP)) {
      waitButtonRelease(BTN_UP);
      currentMode = (currentMode == MODE_INFO) ? MODE_CALENDAR : (currentMode + 1) % DIAL_MODE_COUNT;
      exitReason = EXIT_MODE_CHANGE; return;
    }
    if (isButtonPressed(BTN_DOWN)) {
      waitButtonRelease(BTN_DOWN);
      currentMode = (currentMode == MODE_INFO) ? MODE_AIRENV : (currentMode - 1 + DIAL_MODE_COUNT) % DIAL_MODE_COUNT;
      exitReason = EXIT_MODE_CHANGE; return;
    }
    if (isButtonPressed(BTN_OK)) { waitButtonRelease(BTN_OK); refreshDisplay(); resetActivityTimer(); }
    if (isButtonPressed(BTN_MENU)) { waitButtonRelease(BTN_MENU); }
    delay(50);
  }
  currentMode = MODE_CALENDAR; exitReason = EXIT_MODE_CHANGE;
}

void activeLoop() {
  resetActivityTimer();
  while (!isTimeoutReached()) {
    if (isButtonPressed(BTN_MENU)) {
      if (currentMode == MODE_CALENDAR) {
        calendarView = (calendarView == 0) ? 1 : 0;
        refreshDisplay();
      }
      waitButtonRelease(BTN_MENU); resetActivityTimer();
    }
    if (isButtonPressed(BTN_UP)) {
      waitButtonRelease(BTN_UP);
      currentMode = (currentMode + 1) % DIAL_MODE_COUNT;
      exitReason = EXIT_MODE_CHANGE; return;
    }
    if (isButtonPressed(BTN_DOWN)) {
      waitButtonRelease(BTN_DOWN);
      currentMode = (currentMode - 1 + DIAL_MODE_COUNT) % DIAL_MODE_COUNT;
      exitReason = EXIT_MODE_CHANGE; return;
    }
    if (isButtonPressed(BTN_OK)) { refreshDisplay(); waitButtonRelease(BTN_OK); resetActivityTimer(); }
    if (isButtonPressed(BTN_EXIT)) {
      waitButtonRelease(BTN_EXIT);
      currentMode = MODE_INFO; exitReason = EXIT_MODE_CHANGE; return;
    }
    delay(50);
  }
  exitReason = EXIT_TIMEOUT;
}

void modeDispatcher() {
  while (true) {
    exitReason = EXIT_TIMEOUT;
    refreshDisplay();
    switch (currentMode) {
      case MODE_CALENDAR:
      case MODE_SENSOR:
      case MODE_AIRENV:
        activeLoop();
        if (exitReason == EXIT_TIMEOUT) return;
        break;
      case MODE_INFO:
        infoLoop();
        break;
    }
    if (exitReason == EXIT_TIMEOUT) return;
  }
}

void setup() {
  Serial.begin(115200);
  bootCount++;
  Serial.printf("Boot #%d Mode:%d\n", bootCount, currentMode);

  pinMode(BTN_MENU, INPUT);
  pinMode(BTN_EXIT, INPUT);
  pinMode(BTN_UP, INPUT);
  pinMode(BTN_DOWN, INPUT);
  pinMode(BTN_OK, INPUT);

  // 채널은 마지막 성공 채널을 우선 사용 (첫 부팅이면 기본값)
  if (savedChannel < CH_MIN || savedChannel > CH_MAX) savedChannel = DEFAULT_CHANNEL;
  initESPNow();

  esp_sleep_wakeup_cause_t wakeup = esp_sleep_get_wakeup_cause();
  Serial.printf("Wakeup: %d\n", wakeup);

  // 타이머(정기 기상)로 깨어났으면 캘린더 모드로 시작
  if (wakeup == ESP_SLEEP_WAKEUP_TIMER) {
    currentMode = MODE_CALENDAR;
  }

  modeDispatcher();
  enterDeepSleep();
}
