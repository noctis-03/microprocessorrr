A: CrowPanel 4.2" E-Paper B: Heltec WiFi LoRa V3 ESP-NOW I2S Audio
⚠️ Heltec WiFi LoRa V3 = ESP32-S3 — 중요 제약사항

ESP32-S3는 Bluetooth Classic (A2DP)를 지원하지 않습니다. 따라서 BT 스피커로 직접 오디오 전송이 불가능합니다.
이 가이드에서는 I2S DAC (MAX98357A) + 유선 스피커로 오디오를 출력합니다. Spotify 음악은 핸드폰에서 재생하고, B는 리모컨 역할만 합니다.

RTC_DATA_ATTR int savedYear  = 2026;
RTC_DATA_ATTR int savedMonth = 6;
RTC_DATA_ATTR int savedDay   = 1;

const char* weekdays[] = {"Sun","Mon","Tue","Wed","Thu","Fri","Sat"};

                           "Jul","Aug","Sep","Oct","Nov","Dec"};

void epdPower(bool on) {
  pinMode(EPD_PWR, OUTPUT);
  digitalWrite(EPD_PWR, on ? HIGH : LOW);
}

int getDayOfWeek(int y, int m, int d) {
  struct tm t = {0};
  t.tm_year = y - 1900;  t.tm_mon = m - 1;  t.tm_mday = d;
  mktime(&t);
  return t.tm_wday;
}

void incrementDate() {
  int dim[] = {31,28,31,30,31,30,31,31,30,31,30,31};
  if (savedYear % 4 == 0) dim[1] = 29;
  savedDay++;
  if (savedDay > dim[savedMonth - 1]) {
    savedDay = 1;  savedMonth++;
    if (savedMonth > 12) { savedMonth = 1; savedYear++; }
  }
}

void drawCalendar() {
  epdPower(true);
  display.init(115200, true, 50, false);
  display.setRotation(0);
  display.setFullWindow();
  display.fillScreen(GxEPD_WHITE);

  int dow = getDayOfWeek(savedYear, savedMonth, savedDay);

  // 년도
  display.setFont(&FreeMonoBold24pt7b);
  display.setTextColor(GxEPD_BLACK);
  display.setCursor(130, 55);
  display.print(savedYear);

  display.drawLine(30, 70, 370, 70, GxEPD_BLACK);

  // 월 + 일
  display.setCursor(60, 150);
  char buf[32];
  sprintf(buf, "%s %d", months[savedMonth - 1], savedDay);
  display.print(buf);

  // 요일
  display.setFont(&FreeMonoBold18pt7b);
  display.setCursor(130, 200);
  display.print(weekdays[dow]);

void setup() {
  bootCount++;
  esp_sleep_wakeup_cause_t cause = esp_sleep_get_wakeup_cause();
  if (cause == ESP_SLEEP_WAKEUP_TIMER) incrementDate();

2-4 센서 읽기 코드
#include <Wire.h>
#include <Adafruit_BME280.h>
#include <BH1750.h>

#define I2C_SDA 41
#define I2C_SCL 42
#define PIR_PIN 2

Adafruit_BME280 bme;
BH1750 lightMeter;

struct SensorData {
  float temperature;
  float humidity;
  float pressure;
  float lux;
  bool  motionDetected;
} sensorData;

void initSensors() {
  Wire.begin(I2C_SDA, I2C_SCL);
  if (!bme.begin(0x76, &Wire))
    Serial.println("BME280 not found!");
  if (!lightMeter.begin(BH1750::CONTINUOUS_HIGH_RES_MODE, 0x23, &Wire))
    Serial.println("BH1750 not found!");
  pinMode(PIR_PIN, INPUT);
}

void readSensors() {
  sensorData.temperature    = bme.readTemperature();
  sensorData.humidity       = bme.readHumidity();
  sensorData.pressure       = bme.readPressure() / 100.0F;
  sensorData.lux            = lightMeter.readLightLevel();
  sensorData.motionDetected = digitalRead(PIR_PIN);

  Serial.printf("T:%.1f°C  H:%.0f%%  Lux:%.0f  Motion:%s\n",
    sensorData.temperature, sensorData.humidity,
    sensorData.lux, sensorData.motionDetected ? "YES" : "NO");
}

void setup() {
  Serial.begin(115200);
  initSensors();
}

void loop() {
  readSensors();
  delay(5000);
}

3-1 공통 메시지 구조 (shared_types.h)

A와 B 양쪽 프로젝트에 동일한 헤더 파일을 사용합니다.

#pragma once
#include <stdint.h>

enum MsgType : uint8_t {
  // B → A
  MSG_CALENDAR = 0x01,  MSG_MUSIC   = 0x02,
  MSG_WEATHER  = 0x03,  MSG_NEWS    = 0x04,
  MSG_SENSOR   = 0x05,  MSG_PRESET  = 0x10,
  MSG_TIME_SYNC = 0x11,
  // A → B
  MSG_BUTTON   = 0x20,  MSG_REQUEST = 0x21,
  MSG_ACK      = 0xFF
};

typedef struct { MsgType type; uint8_t id; } MsgHeader;

typedef struct {
  MsgHeader header;
  int year, month, day, hour, minute;
} CalendarMsg;

typedef struct {
  MsgHeader header;
  char title[64];  char artist[48];
  bool isPlaying;  uint16_t progressPct;
} MusicMsg;

typedef struct {
  MsgHeader header;
  int8_t temp, tempHigh, tempLow;
  uint8_t humidity, icon;
  char desc[32];
} WeatherMsg;

typedef struct {
  MsgHeader header;
  float temperature, humidity, lux;
  bool motion;
} SensorMsg;

typedef struct { MsgHeader header; uint8_t preset; } PresetMsg;
typedef struct { MsgHeader header; uint8_t buttonId; } ButtonMsg;
3-2 유선 Wake-up 배선
물리 연결 (점퍼 와이어 2개): B (GPIO 26) ──────────── A (GPIO 3) B (GND) ──────────── A (GND) A 측: GPIO 3에 10KΩ 풀다운 저항 (GPIO3 → GND) → 평소 LOW 유지, B가 HIGH 보내면 A wake
💡 왜 GPIO 3인가?

ESP32-S3의 RTC GPIO 범위는 0~21입니다. CrowPanel에서 GPIO 3은 GPIO 확장 포트로 접근 가능하며, 버튼과 겹치지 않습니다.

3-3 B 측: ESP-NOW 송신 코드
#include <esp_now.h>
#include <WiFi.h>
#include "shared_types.h"

#define WAKE_PIN 26

// ★ A의 MAC 주소 (A 시리얼 모니터에서 WiFi.macAddress() 로 확인)
uint8_t peerA[] = {0xAA, 0xBB, 0xCC, 0xDD, 0xEE, 0xFF};

void initESPNOW_B() {
  WiFi.mode(WIFI_STA);
  esp_now_init();
  esp_now_peer_info_t pi = {};
  memcpy(pi.peer_addr, peerA, 6);
  pi.channel = 0;  pi.encrypt = false;
  esp_now_add_peer(&pi);
}

void wakeA() {
  pinMode(WAKE_PIN, OUTPUT);
  digitalWrite(WAKE_PIN, HIGH);
  delay(10);
  digitalWrite(WAKE_PIN, LOW);
  delay(500);   // A 부팅 대기
}

void sendSensorToA() {
  wakeA();
  SensorMsg msg;
  msg.header = { MSG_SENSOR, (uint8_t)(millis() & 0xFF) };
  msg.temperature = sensorData.temperature;
  msg.humidity    = sensorData.humidity;
  msg.lux         = sensorData.lux;
  msg.motion      = sensorData.motionDetected;
  esp_now_send(peerA, (uint8_t*)&msg, sizeof(msg));
}
3-4 A 측: ESP-NOW 수신 + 절전 프리셋
#include <esp_now.h>
#include <WiFi.h>
#include "shared_types.h"

#define WIRE_WAKE 3     // B에서 들어오는 wake 신호

RTC_DATA_ATTR uint8_t currentPreset = 1;   // 기본: 스마트
volatile bool dataReceived = false;
uint8_t recvBuf[256];
int recvLen = 0;

void onRecv(const uint8_t* mac, const uint8_t* data, int len) {
  memcpy(recvBuf, data, len);
  recvLen = len;
  dataReceived = true;
}

void initESPNOW_A() {
  WiFi.mode(WIFI_STA);
  esp_now_init();
  esp_now_register_recv_cb(onRecv);
  // B를 peer로 등록
  uint8_t peerB[] = {0x11,0x22,0x33,0x44,0x55,0x66};
  esp_now_peer_info_t pi = {};
  memcpy(pi.peer_addr, peerB, 6);
  esp_now_add_peer(&pi);
}

void waitForData(uint32_t ms) {
  unsigned long t = millis();
  while (!dataReceived && millis() - t < ms) delay(10);
}

void processData() {
  MsgHeader* h = (MsgHeader*)recvBuf;
  switch (h->type) {
    case MSG_CALENDAR: {
      CalendarMsg* m = (CalendarMsg*)recvBuf;
      savedYear = m->year; savedMonth = m->month; savedDay = m->day;
      drawCalendar();
      break;
    }
    case MSG_SENSOR: {
      SensorMsg* m = (SensorMsg*)recvBuf;
      drawSensorScreen(m->temperature, m->humidity, m->lux);
      break;
    }
    case MSG_WEATHER: {
      WeatherMsg* m = (WeatherMsg*)recvBuf;
      drawWeatherScreen(m->temp, m->humidity, m->desc);
      break;
    }
    case MSG_MUSIC: {
      MusicMsg* m = (MusicMsg*)recvBuf;
      drawMusicScreen(m->title, m->artist, m->isPlaying);
      break;
    }
    case MSG_PRESET: {
      PresetMsg* m = (PresetMsg*)recvBuf;
      currentPreset = m->preset;
      break;
    }
  }
}

// ★ 4단계 절전 프리셋
void enterSleep() {
  esp_now_deinit();
  WiFi.mode(WIFI_OFF);
  display.hibernate();
  epdPower(false);

  switch (currentPreset) {
    case 0:  // 액자 — 24시간
      esp_sleep_enable_timer_wakeup(24ULL * 3600 * 1000000);  break;
    case 1:  // 스마트 — 5분
      esp_sleep_enable_timer_wakeup(5ULL * 60 * 1000000);     break;
    case 2:  // 뮤직 — 30초
      esp_sleep_enable_timer_wakeup(30ULL * 1000000);          break;
    case 3:  // 라이브 — 5초
      esp_sleep_enable_timer_wakeup(5ULL * 1000000);           break;
  }
  // 유선 wake (B → A)
  esp_sleep_enable_ext1_wakeup(
    (1ULL << WIRE_WAKE), ESP_EXT1_WAKEUP_ANY_HIGH);
  // 버튼 wake
  esp_sleep_enable_ext0_wakeup(GPIO_NUM_2, LOW);

  esp_deep_sleep_start();
}

void setup() {
  bootCount++;
  esp_sleep_wakeup_cause_t cause = esp_sleep_get_wakeup_cause();

  initESPNOW_A();

  switch (cause) {
    case ESP_SLEEP_WAKEUP_TIMER:
      sendRequestToB();
      waitForData(2000);
      if (dataReceived) processData(); else drawCalendar();
      break;
    case ESP_SLEEP_WAKEUP_EXT1:   // B의 GPIO 신호
      waitForData(3000);
      if (dataReceived) processData();
      break;
    case ESP_SLEEP_WAKEUP_EXT0:   // 사용자 버튼
      handleButtonMenu();
      break;
    default:
      drawCalendar();
      break;
  }
  enterSleep();
}

void loop() {}

String spotifyToken = "";
String refreshToken = "YOUR_REFRESH_TOKEN";

void refreshSpotifyToken() {
  HTTPClient http;
  http.begin("https://accounts.spotify.com/api/token");
  http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    JsonDocument doc; deserializeJson(doc, http.getString());
    spotifyToken = doc["access_token"].as<String>();

  http.end();

void getSpotifyNowPlaying() {
  HTTPClient http;
  http.begin("https://api.spotify.com/v1/me/player/currently-playing");
  http.addHeader("Authorization", "Bearer " + spotifyToken);
  int code = http.GET();
  if (code == 200) {
    JsonDocument doc; deserializeJson(doc, http.getString());
    MusicMsg msg;
    msg.header = { MSG_MUSIC, 0 };
    msg.isPlaying = doc["is_playing"];
    msg.progressPct = (uint16_t)(
      doc["progress_ms"].as<float>() /
      doc["item"]["duration_ms"].as<float>() * 1000);
    strncpy(msg.title,  doc["item"]["name"]                 | "", 63);
    strncpy(msg.artist, doc["item"]["artists"][0]["name"]   | "", 47);
    wakeA();
    esp_now_send(peerA, (uint8_t*)&msg, sizeof(msg));
  } else if (code == 401) {
    refreshSpotifyToken();
  }
  http.end();
}

5-1 MAX98357A 배선
MAX98357A I2S 앰프 모듈: VIN → Heltec 3V3 (또는 5V) GND → GND BCLK → GPIO 47 LRC → GPIO 48 DIN → GPIO 38 SD → 연결 안 함 (또는 VIN에 풀업 = L+R 합산) 스피커: + → MAX98357A (+) - → MAX98357A (-)
💡 PSRAM 없음 주의

Heltec V3에는 PSRAM이 없습니다. 오디오 버퍼를 작게 유지하고, 낮은 비트레이트 스트림(64~128kbps)을 사용하세요.

5-2 인터넷 라디오 스트리밍

ESP8266Audio 라이브러리를 사용합니다. (Library Manager에서 설치)

#include "AudioGeneratorMP3.h"
#include "AudioOutputI2S.h"
#include "AudioFileSourceHTTPStream.h"
#include "AudioFileSourceBuffer.h"

AudioGeneratorMP3        *mp3      = nullptr;
AudioOutputI2S           *audioOut = nullptr;
AudioFileSourceHTTPStream *stream  = nullptr;
AudioFileSourceBuffer    *buff     = nullptr;

const char* radioStations[] = {
  "http://stream.live.vc.bbcmedia.co.uk/bbc_radio_one",
  "http://ice1.somafm.com/groovesalad-128-mp3",
  "http://streams.kqed.org/kqedradio"
};
int numStations = 3;
int currentStation = 0;

void initAudio() {
  audioOut = new AudioOutputI2S();
  audioOut->SetPinout(47, 48, 38);  // BCLK, LRC, DOUT
  audioOut->SetGain(0.3);
}

void playRadio(int idx) {
  stopRadio();
  currentStation = idx % numStations;
  stream = new AudioFileSourceHTTPStream(radioStations[currentStation]);
  buff   = new AudioFileSourceBuffer(stream, 2048);  // 작은 버퍼
  mp3    = new AudioGeneratorMP3();
  mp3->begin(buff, audioOut);
  Serial.printf("Playing station %d\n", currentStation);
}

void stopRadio() {
  if (mp3 && mp3->isRunning()) mp3->stop();
  delete mp3;    mp3    = nullptr;
  delete buff;   buff   = nullptr;
  delete stream; stream = nullptr;
}

void nextStation() {
  playRadio(currentStation + 1);
}

// ★ loop() 또는 FreeRTOS 태스크에서 반복 호출 필수
void audioLoop() {
  if (mp3 && mp3->isRunning()) {
    if (!mp3->loop()) {
      mp3->stop();
      Serial.println("Stream ended");
    }
  }
}

7-2 자동 스케줄러
void autoScheduler() {
  static unsigned long lastWeather = 0, lastSpotify = 0, lastNTP = 0;
  unsigned long now = millis();

  // 30분마다 날씨 갱신 → A에 전송
  if (now - lastWeather > 30UL * 60 * 1000) {
    fetchWeather();  lastWeather = now;
  }
  // 뮤직 모드일 때 30초마다 Spotify 확인
  if (currentPreset == 2 && now - lastSpotify > 30000) {
    getSpotifyNowPlaying();  lastSpotify = now;
  }
  // 6시간마다 NTP 재동기화
  if (now - lastNTP > 6UL * 3600 * 1000) {
    syncNTP();  sendTimeToA();  lastNTP = now;
  }
}
7-3 자동 모드 전환 로직
void autoModeCheck() {
  struct tm ti;  getLocalTime(&ti);

  // 야간 (23시~6시) + PIR 미감지 → 액자
  if ((ti.tm_hour >= 23 || ti.tm_hour < 6)
      && !sensorData.motionDetected
      && currentPreset != 0) {
    sendPresetToA(0);  currentPreset = 0;
  }

  // 라디오 재생 중 → 뮤직 모드
  bool playing = mp3 && mp3->isRunning();
  static unsigned long stoppedAt = 0;

  if (playing && currentPreset != 2) {
    sendPresetToA(2);  currentPreset = 2;  stoppedAt = 0;
  }

  // 라디오 종료 10분 후 → 스마트 모드 복귀
  if (!playing && currentPreset == 2) {
    if (stoppedAt == 0) stoppedAt = millis();
    if (millis() - stoppedAt > 10UL * 60 * 1000) {
      sendPresetToA(1);  currentPreset = 1;  stoppedAt = 0;
    }
  }
}

void sendPresetToA(uint8_t p) {
  PresetMsg msg = {{ MSG_PRESET, 0 }, p};
  wakeA();
  esp_now_send(peerA, (uint8_t*)&msg, sizeof(msg));
}

bool sendWithRetry(const uint8_t* peer, const uint8_t* data,
                   size_t len, int maxRetries = 3) {
  for (int i = 0; i < maxRetries; i++) {
    esp_err_t result = esp_now_send(peer, data, len);
    if (result == ESP_OK) return true;
    delay(50 * (i + 1));  // 점진적 대기
  }
  Serial.println("ESP-NOW send failed after retries");
  return false;
}
