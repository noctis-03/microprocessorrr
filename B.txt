// ══════════════════════════════════════
// 센서 읽기
// ══════════════════════════════════════
void readSensors() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  if (!isnan(t)) temperature = t;
  if (!isnan(h)) humidity = h;
  lightLevel = analogRead(PIN_LIGHT);
  if (digitalRead(PIN_PIR) == HIGH) {
    motionState = true; lastMotionTime = millis();
  } else if (millis() - lastMotionTime > MOTION_HOLD_MS) {
    motionState = false;
  }
}

bool getTime(struct tm *info) { return getLocalTime(info, 1000); }

// ══════════════════════════════════════
// 날씨 API (Open-Meteo)
// ══════════════════════════════════════
bool fetchWeather() {
  if (WiFi.status() != WL_CONNECTED) return false;

  char url[360];
  snprintf(url, sizeof(url),
    "http://api.open-meteo.com/v1/forecast?latitude=%.4f&longitude=%.4f"
    "&hourly=temperature_2m,relative_humidity_2m,weather_code,precipitation_probability"
    "&daily=temperature_2m_max,temperature_2m_min"
    "&timezone=Asia%%2FSeoul&forecast_days=1",
    g_cfg.lat, g_cfg.lon);

  HTTPClient http;
  http.begin(url);
  int code = http.GET();
  if (code != 200) { Serial.printf("[Weather] HTTP %d\n", code); http.end(); return false; }

  String payload = http.getString();
  http.end();

  StaticJsonDocument<384> filter;
  filter["hourly"]["temperature_2m"] = true;
  filter["hourly"]["relative_humidity_2m"] = true;
  filter["hourly"]["weather_code"] = true;
  filter["hourly"]["precipitation_probability"] = true;
  filter["daily"]["temperature_2m_max"] = true;
  filter["daily"]["temperature_2m_min"] = true;

  DynamicJsonDocument doc(10240);
  DeserializationError err = deserializeJson(doc, payload,
                                DeserializationOption::Filter(filter));
  if (err) { Serial.printf("[Weather] JSON err: %s\n", err.c_str()); return false; }

  JsonArray temps = doc["hourly"]["temperature_2m"];
  JsonArray hums  = doc["hourly"]["relative_humidity_2m"];
  JsonArray codes = doc["hourly"]["weather_code"];
  JsonArray pops  = doc["hourly"]["precipitation_probability"];

  if (temps.isNull() || temps.size() == 0) { Serial.println("[Weather] empty"); return false; }

  uint8_t maxPop = 0;
  for (int i = 0; i < WEATHER_SLOTS; i++) {
    int idx = i * 4;
    if (idx >= (int)temps.size()) { g_fcHour[i] = -1; continue; }
    g_fcHour[i]  = idx;
    g_fcTemp[i]  = (int8_t)lround((double)temps[idx]);
    g_fcHumid[i] = (uint8_t)(int)hums[idx];
    g_fcCode[i]  = (uint8_t)(int)codes[idx];
  }
  for (int i = 0; i < (int)pops.size(); i++) {
    int p = (int)pops[i];
    if (p > maxPop) maxPop = p;
  }
  g_todayPop = maxPop;

  JsonArray tmax = doc["daily"]["temperature_2m_max"];
  JsonArray tmin = doc["daily"]["temperature_2m_min"];
  if (!tmax.isNull() && tmax.size() > 0) g_todayMax = (int8_t)lround((double)tmax[0]);
  if (!tmin.isNull() && tmin.size() > 0) g_todayMin = (int8_t)lround((double)tmin[0]);

  g_weatherValid = true;
  g_lastWeatherFetch = millis();
  Serial.printf("[Weather] OK  max%d min%d pop%d\n", g_todayMax, g_todayMin, g_todayPop);
  return true;
}

// ══════════════════════════════════════
// 대기질 API (Open-Meteo Air Quality)
// ══════════════════════════════════════
bool fetchAirQuality() {
  if (WiFi.status() != WL_CONNECTED) return false;

  char url[320];
  snprintf(url, sizeof(url),
    "http://air-quality-api.open-meteo.com/v1/air-quality?latitude=%.4f&longitude=%.4f"
    "&hourly=pm2_5,us_aqi&timezone=Asia%%2FSeoul&forecast_days=1",
    g_cfg.lat, g_cfg.lon);

  HTTPClient http;
  http.begin(url);
  int code = http.GET();
  if (code != 200) { Serial.printf("[Air] HTTP %d\n", code); http.end(); return false; }

  String payload = http.getString();
  http.end();

  StaticJsonDocument<256> filter;
  filter["hourly"]["pm2_5"] = true;
  filter["hourly"]["us_aqi"] = true;

  DynamicJsonDocument doc(8192);
  DeserializationError err = deserializeJson(doc, payload,
                                DeserializationOption::Filter(filter));
  if (err) { Serial.printf("[Air] JSON err: %s\n", err.c_str()); return false; }

  JsonArray pm  = doc["hourly"]["pm2_5"];
  JsonArray aqi = doc["hourly"]["us_aqi"];
  if (pm.isNull() || pm.size() == 0) { Serial.println("[Air] empty"); return false; }

  int idx = 0;
  struct tm ti;
  if (timeSync && getTime(&ti)) idx = ti.tm_hour;
  if (idx >= (int)pm.size()) idx = 0;

  g_pm25 = (uint8_t)constrain((int)lround((double)pm[idx]), 0, 255);
  if (!aqi.isNull() && idx < (int)aqi.size())
    g_usAqi = (uint8_t)constrain((int)aqi[idx], 0, 255);

  g_envValid = true;
  Serial.printf("[Air] OK  PM2.5:%d  AQI:%d\n", g_pm25, g_usAqi);
  return true;
}

// ══════════════════════════════════════
// AQI 등급
// ══════════════════════════════════════
const char* aqiGradeStr(uint8_t aqi) {
  if (aqi == 0)   return "Unknown";
  if (aqi <= 50)  return "Good";
  if (aqi <= 100) return "Moderate";
  if (aqi <= 150) return "Unhealthy for sensitive groups";
  if (aqi <= 200) return "Unhealthy";
  if (aqi <= 300) return "Very unhealthy";
  return "Hazardous";
}

// ══════════════════════════════════════
// Gemini 호출 → 오늘의 영어 멘트 생성
// ══════════════════════════════════════
bool fetchAIComment() {
  if (WiFi.status() != WL_CONNECTED) return false;

  char prompt[512];
  snprintf(prompt, sizeof(prompt),
    "You are a friendly daily assistant for a desk display in Daejeon, Korea. "
    "Today's data: high %d C, low %d C, rain chance %d%%, "
    "PM2.5 %d ug/m3, US AQI %d (%s). "
    "Write ONE short practical life tip in English, max 18 words, "
    "plain text only, no quotes, no emoji.",
    g_todayMax, g_todayMin, g_todayPop, g_pm25, g_usAqi, aqiGradeStr(g_usAqi));

  DynamicJsonDocument reqDoc(1024);
  JsonObject part = reqDoc["contents"][0]["parts"][0].to<JsonObject>();
  part["text"] = prompt;
  reqDoc["generationConfig"]["maxOutputTokens"] = 60;
  reqDoc["generationConfig"]["temperature"] = 0.7;

  String reqBody;
  serializeJson(reqDoc, reqBody);

  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient https;
  char url[160];
  snprintf(url, sizeof(url),
    "https://generativelanguage.googleapis.com/v1beta/models/%s:generateContent",
    GEMINI_MODEL);

  if (!https.begin(client, url)) { Serial.println("[AI] begin fail"); return false; }
  https.addHeader("Content-Type", "application/json");
  https.addHeader("x-goog-api-key", GEMINI_API_KEY);

  int code = https.POST(reqBody);
  if (code != 200) {
    Serial.printf("[AI] HTTP %d\n", code);
    Serial.println(https.getString());
    https.end();
    return false;
  }

  String payload = https.getString();
  https.end();

  StaticJsonDocument<256> filter;
  filter["candidates"][0]["content"]["parts"][0]["text"] = true;

  DynamicJsonDocument doc(4096);
  DeserializationError err = deserializeJson(doc, payload,
                                DeserializationOption::Filter(filter));
  if (err) { Serial.printf("[AI] JSON err: %s\n", err.c_str()); return false; }

  const char* text = doc["candidates"][0]["content"]["parts"][0]["text"];
  if (!text) { Serial.println("[AI] no text in response"); return false; }

  String s = String(text);
  s.trim();
  s.replace("\n", " ");
  s.replace("\r", " ");
  strncpy(g_aiComment, s.c_str(), AI_BUF_SIZE - 1);
  g_aiComment[AI_BUF_SIZE - 1] = '\0';

  Serial.printf("[AI] comment: %s\n", g_aiComment);
  return true;
}

// ══════════════════════════════════════
// 오늘 멘트 확보
// ══════════════════════════════════════
void ensureTodayComment() {
  if (!g_cfg.aiEnabled) return;          // ★ AI 비활성화 시 생성 안 함
  if (!timeSync) return;
  if (WiFi.status() != WL_CONNECTED) return;

  struct tm ti;
  if (!getTime(&ti)) return;
  int today = ti.tm_mday;

  if (g_aiCommentValid && g_aiCommentDay == today) return;

  if (!g_weatherValid) fetchWeather();
  if (!g_envValid)     fetchAirQuality();

   if (!g_weatherValid) {          // ★ 추가: 날씨 확보 실패 시 멘트 생성 보류
    Serial.println("[AI] weather not ready, skip comment");
    return;
  }

  if (fetchAIComment()) {
    g_aiCommentDay = today;
    g_aiCommentValid = true;
    saveCommentToNVS();
    Serial.printf("[AI] today's comment ready (day %d)\n", today);
  }
}

// ══════════════════════════════════════
// AI 멘트를 chunk로 나눠 A에 전송
// ══════════════════════════════════════
void sendAIComment() {
  int len = strlen(g_aiComment);
  uint8_t total = (len + AI_CHUNK_SIZE - 1) / AI_CHUNK_SIZE;
  if (total == 0) total = 1;
  if (total > AI_MAX_CHUNKS) total = AI_MAX_CHUNKS;

  for (uint8_t i = 0; i < total; i++) {
    AIChunk ck = {};
    ck.msgType = MSG_AI_CHUNK;
    ck.totalChunks = total;
    ck.chunkIndex = i;
    ck.commentDay = (uint8_t)(g_aiCommentDay < 0 ? 0 : g_aiCommentDay);

    int off = i * AI_CHUNK_SIZE;
    int remain = len - off;
    int n = remain > AI_CHUNK_SIZE ? AI_CHUNK_SIZE : remain;
    if (n < 0) n = 0;
    ck.dataLen = n;
    memcpy(ck.data, g_aiComment + off, n);

    esp_now_send(peerMAC_A, (uint8_t *)&ck, sizeof(AIChunk));
    delay(20);
  }
  Serial.printf("[AI] sent %d chunk(s) to A\n", total);
}

// ══════════════════════════════════════
// ESP‑NOW 수신 콜백
// ══════════════════════════════════════
void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int data_len) {
  if (data_len == sizeof(Message)) {
    Message incoming;
    memcpy(&incoming, data, sizeof(Message));
    if (incoming.msgType != MSG_REQUEST) return;

    Serial.println("[B] Sensor request from A");
    readSensors();

    Message response = {};
    response.msgType = MSG_SENSOR_DATA;
    response.temperature = temperature;
    response.humidity = humidity;

    struct tm timeinfo;
    if (timeSync && getTime(&timeinfo)) {
      response.year = timeinfo.tm_year + 1900;
      response.month = timeinfo.tm_mon + 1;
      response.day = timeinfo.tm_mday;
      response.weekday = timeinfo.tm_wday;
      response.hour = timeinfo.tm_hour;
      response.minute = timeinfo.tm_min;
    }

    if (g_weatherValid) {
      memcpy(response.fcHour,  g_fcHour,  sizeof(g_fcHour));
      memcpy(response.fcTemp,  g_fcTemp,  sizeof(g_fcTemp));
      memcpy(response.fcHumid, g_fcHumid, sizeof(g_fcHumid));
      memcpy(response.fcCode,  g_fcCode,  sizeof(g_fcCode));
    } else {
      memset(response.fcHour, -1, sizeof(response.fcHour));
    }

    response.todayMax = g_todayMax;
    response.todayMin = g_todayMin;
    response.todayPop = g_todayPop;
    response.pm25     = g_pm25;
    response.usAqi    = g_usAqi;
    response.config   = g_cfg;        // ★ 현재 설정값을 A에게 전달

    esp_now_send(peerMAC_A, (uint8_t *)&response, sizeof(Message));

    requestReceived = true;
    requestDisplayTime = millis();
    Serial.printf("[B] Sent sensor (weather:%s air:%s cfgVer:%d)\n",
                  g_weatherValid ? "Y" : "N", g_envValid ? "Y" : "N", g_cfg.configVersion);
    return;
  }

  if (data_len == sizeof(AIChunk)) {
    AIChunk req;
    memcpy(&req, data, sizeof(AIChunk));
    if (req.msgType != MSG_AI_REQUEST) return;

    if (!g_cfg.aiEnabled) {           // ★ AI 비활성화 시
      Serial.println("[B] AI disabled by config");
      return;
    }
    Serial.println("[B] AI comment request from A");
    ensureTodayComment();
    if (g_aiCommentValid && g_aiComment[0] != '\0') {
      sendAIComment();
    } else {
      Serial.println("[B] no comment yet -> A shows 'Requesting AI...'");
    }
    return;
  }

  // ── C 보드 요청 처리 ──
  if (data_len == sizeof(CRequest)) {
    CRequest req;
    memcpy(&req, data, sizeof(CRequest));
    if (req.msgType != MSG_C_REQUEST) return;

    readSensors();   // 최신 온도/조도/PIR 확보

    CMessage resp = {};
    resp.msgType     = MSG_C_DATA;
    resp.temperature = temperature;
    resp.lightLevel  = lightLevel;
    resp.motion      = motionState ? 1 : 0;
    resp.config      = g_cCfg;

    esp_now_send(peerMAC_C, (uint8_t *)&resp, sizeof(CMessage));
    Serial.printf("[B->C] T=%.1f L=%d PIR=%d mode=%d out=%d ver=%d\n",
                  temperature, lightLevel, motionState,
                  g_cCfg.cMode, g_cCfg.cOutput, g_cCfg.cfgVersion);
    return;
  }
}

void onDataSent(const wifi_tx_info_t *info, esp_now_send_status_t status) {
  Serial.printf("[B SEND] %s\n", status == ESP_NOW_SEND_SUCCESS ? "OK" : "FAIL");
}

// ══════════════════════════════════════
// OLED
// ══════════════════════════════════════
void updateOLED() {
  oled.clearDisplay();
  oled.setTextColor(WHITE);
  oled.setTextSize(1);

  oled.setCursor(0, 0);
  struct tm timeinfo;
  if (timeSync && getTime(&timeinfo)) {
    oled.printf("%04d-%02d-%02d %02d:%02d:%02d",
               timeinfo.tm_year + 1900, timeinfo.tm_mon + 1, timeinfo.tm_mday,
               timeinfo.tm_hour, timeinfo.tm_min, timeinfo.tm_sec);
  } else {
    oled.print("Time: not synced");
  }

  oled.setCursor(0, 14);
  oled.printf("Temp : %.1f C", temperature);
  oled.setCursor(0, 26);
  oled.printf("Humid: %.1f %%", humidity);
  oled.setCursor(0, 38);
  oled.printf("L:%d PIR:%d C:%s", lightLevel, motionState,
              g_cCfg.cMode == CMODE_AUTO ? "AUTO" : "DEF");

  oled.setCursor(0, 52);
  if (requestReceived && (millis() - requestDisplayTime < REQUEST_DISPLAY_MS)) {
    oled.print("> Sent to A!");
  } else {
    requestReceived = false;
    oled.print(statusLine.c_str());
  }
  oled.display();
}

// ══════════════════════════════════════
// 웹: 설정 저장
// ══════════════════════════════════════
void handleSave() {
  bool wifiChanged = false;

  if (server.hasArg("ssid") && server.arg("ssid").length() > 0) {
    String ssid = server.arg("ssid");
    String pass = server.arg("pass");
    if (ssid != savedSSID) wifiChanged = true;
    prefs.begin("wifi", false);
    prefs.putString("ssid", ssid);
    if (pass.length() > 0) prefs.putString("pass", pass);  // 빈칸이면 기존 유지
    prefs.end();
  }

  if (server.hasArg("lat")) g_cfg.lat = server.arg("lat").toFloat();
  if (server.hasArg("lon")) g_cfg.lon = server.arg("lon").toFloat();

  if (server.hasArg("wake"))  g_cfg.wakeHour         = constrain((int)server.arg("wake").toInt(), 0, 23);
  if (server.hasArg("slpto")) g_cfg.sleepTimeoutSec  = constrain((int)server.arg("slpto").toInt(), 3, 120);
  if (server.hasArg("fullr")) g_cfg.fullRefreshEvery = constrain((int)server.arg("fullr").toInt(), 1, 50);
  if (server.hasArg("ai"))    g_cfg.aiEnabled        = server.arg("ai").toInt() ? 1 : 0;

  // ── C 보드 설정 ──
  bool cChanged = false;
  if (server.hasArg("cmode")) {
    uint8_t m = server.arg("cmode").toInt() ? CMODE_AUTO : CMODE_DEFAULT;
    if (m != g_cCfg.cMode) { g_cCfg.cMode = m; cChanged = true; }
  }
  if (server.hasArg("coutput")) {
    uint8_t o = constrain((int)server.arg("coutput").toInt(), 0, 2);
    if (o != g_cCfg.cOutput) { g_cCfg.cOutput = o; cChanged = true; }
  }
  if (cChanged) {
    g_cCfg.cfgVersion++;
    saveCConfigToNVS();
  }

  g_cfg.configVersion++;          // ★ 버전 증가 → A가 "새 설정"으로 인식
  saveConfigToNVS();

  if (wifiConnected) { fetchWeather(); fetchAirQuality(); }

  String msg = "<html><head><meta charset='UTF-8'><meta http-equiv='refresh' content='2;url=/'></head>"
               "<body style='font-family:sans-serif;background:#1a1a2e;color:#eee;padding:20px;'>"
               "<h2>Saved.</h2><p>A board will apply on next wake-up.</p>";
  if (wifiChanged) msg += "<p>WiFi changed - rebooting...</p>";
  msg += "</body></html>";
  server.send(200, "text/html", msg);

  if (wifiChanged) { delay(2000); ESP.restart(); }
}

  if (wifiConnected) { fetchWeather(); fetchAirQuality(); }
  server.send(200, "text/html",
    "<html><head><meta charset='UTF-8'><meta http-equiv='refresh' content='2;url=/'></head>"
    "<body style='font-family:sans-serif;background:#1a1a2e;color:#eee;padding:20px;'>"
    "<h2>Factory defaults restored.</h2></body></html>");
}

// ══════════════════════════════════════
// WiFi 연결
// ══════════════════════════════════════
bool connectWiFi() {
  prefs.begin("wifi", true);
  savedSSID = prefs.getString("ssid", "");
  savedPASS = prefs.getString("pass", "");
  prefs.end();
  if (savedSSID.length() == 0) return false;

  Serial.printf("Connecting to: %s\n", savedSSID.c_str());
  WiFi.mode(WIFI_AP_STA);
  WiFi.begin(savedSSID.c_str(), savedPASS.c_str());

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500); Serial.print("."); attempts++;
  }
  Serial.println();
  if (WiFi.status() == WL_CONNECTED) {
    Serial.printf("Connected! IP: %s CH:%d\n",
                  WiFi.localIP().toString().c_str(), WiFi.channel());
    return true;
  }
  Serial.println("WiFi failed");
  return false;
}

void startAPMode() {
  apMode = true;
  WiFi.mode(WIFI_AP_STA);
  WiFi.softAP(AP_SSID, AP_PASS);
  Serial.printf("AP: %s  http://%s\n", AP_SSID, WiFi.softAPIP().toString().c_str());
  registerRoutes();
  server.begin();
  statusLine = "AP:" + String(AP_SSID);
}

// ══════════════════════════════════════
// ESP‑NOW
// ══════════════════════════════════════
void initESPNow() {
  if (esp_now_init() != ESP_OK) { Serial.println("ESP-NOW init FAILED!"); return; }
  esp_now_register_recv_cb(onDataRecv);
  esp_now_register_send_cb(onDataSent);

  esp_now_peer_info_t peerInfo = {};
  memcpy(peerInfo.peer_addr, peerMAC_A, 6);
  peerInfo.channel = 0;
  peerInfo.ifidx = WIFI_IF_STA;
  peerInfo.encrypt = false;
  esp_now_add_peer(&peerInfo);

  esp_now_peer_info_t peerC = {};
  memcpy(peerC.peer_addr, peerMAC_C, 6);
  peerC.channel = 0;
  peerC.ifidx = WIFI_IF_STA;
  peerC.encrypt = false;
  esp_now_add_peer(&peerC);

  Serial.printf("ESP-NOW ready (B) on CH:%d\n", WiFi.channel());
}

// ══════════════════════════════════════
// Setup
// ══════════════════════════════════════
void setup() {
  Serial.begin(115200);
  delay(1000);

  pinMode(VEXT_PIN, OUTPUT);
  digitalWrite(VEXT_PIN, LOW);
  delay(100);

  Wire.begin(OLED_SDA, OLED_SCL);
  oled.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR);
  oled.clearDisplay();
  oled.display();

  analogReadResolution(12);
  pinMode(PIN_PIR, INPUT);
  dht.begin();

  for (int i = 0; i < WEATHER_SLOTS; i++) g_fcHour[i] = -1;

  loadConfigFromNVS();      // ★ 설정 먼저 (날씨 fetch 전 좌표 필요)
  loadCConfigFromNVS();     // ★ C 보드 설정
  loadCommentFromNVS();

  if (connectWiFi()) {
    wifiConnected = true;
    configTime(GMT_OFFSET, DST_OFFSET, NTP_SERVER);
    struct tm timeinfo;
    if (getLocalTime(&timeinfo, 5000)) {
      timeSync = true;
      statusLine = "WiFi OK | Waiting...";
    } else {
      statusLine = "WiFi OK | NTP fail";
    }

    fetchWeather();
    fetchAirQuality();
    ensureTodayComment();

    registerRoutes();
    server.begin();
  } else {
    startAPMode();
  }

  initESPNow();
  Serial.println("B Board ready");
}

// ══════════════════════════════════════
// Loop
// ══════════════════════════════════════
void loop() {
  readSensors();
  updateOLED();
  server.handleClient();

  if (wifiConnected &&
      (!g_weatherValid || millis() - g_lastWeatherFetch > WEATHER_FETCH_INTERVAL_MS)) {
    fetchWeather();
    fetchAirQuality();
  }

  ensureTodayComment();

  delay(500);
}
